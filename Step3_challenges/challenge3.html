Challenge 3: Dynamic Custom Cursor

Goal: Make your React app more interactive by building a custom customize the cursor so it changes style (e.g. color, shape, size) when hovering over certain elements.
  1. Navigate to your React project
    2. cd my-app
  2. Create a components folder
    Inside the src folder, create a folder called components.
  3. Create two files inside components:
    Cursor.js
    Cursor.css
  4. Add this code to Cursor.js
    5. // src/components/Cursor.js
import React, { useEffect, useState } from "react";
import "./Cursor.css"; // We'll create this CSS file for styling

const Cursor = () => {
  const [position, setPosition] = useState({ x: 0, y: 0 });
  const [hovering, setHovering] = useState(false); // Whether we're over a special area

  // Track mouse movement
  useEffect(() => {
    const move = (e) => {
      setPosition({ x: e.clientX, y: e.clientY });
    };

    window.addEventListener("mousemove", move);

    return () => {
      window.removeEventListener("mousemove", move);
    };
  }, []);

  // Optional: track hover state by listening to custom data attributes
  useEffect(() => {
    const handleMouseOver = (e) => {
      const isHoverTarget = e.target.getAttribute("data-cursor") === "hover";
      setHovering(isHoverTarget);
    };

    window.addEventListener("mouseover", handleMouseOver);

    return () => {
      window.removeEventListener("mouseover", handleMouseOver);
    };
  }, []);

  return (
    <div
      className={`custom-cursor ${hovering ? "hovering" : ""}`}
      style={{ left: `${position.x}px`, top: `${position.y}px` }}
    ></div>
  );
};

export default Cursor;
  5. Add this code to Cursor.css
    6. .custom-cursor {
  position: fixed;
  width: 20px;
  height: 20px;
  border: 2px solid black;
  border-radius: 50%;
  transform: translate(-50%, -50%);
  pointer-events: none; /* So the cursor doesn't block interactions */
  z-index: 9999;
  transition: all 0.2s ease;
  mix-blend-mode: difference; /* Cool effect on light backgrounds */
}

/* When hovering over specific targets */
.custom-cursor.hovering {
  background-color: black;
  width: 40px;
  height: 40px;
  border: none;
}
  6. Update your App.js
    Comment out or delete your current App.js and add this code: 
    7. // src/App.js
import React from "react";
import Cursor from "./components/Cursor";
import "./App.css";

function App() {
  return (
    <div className="App">
      {/* Load our custom cursor */}
      <Cursor />

      <h1>Custom Cursor Demo</h1>

      <div style={{ marginTop: "50px" }}>
        <button data-cursor="hover">Hover over me</button>
      </div>

      <div style={{ marginTop: "50px" }}>
        <p>
          Normal paragraph. Cursor should be small here.
        </p>
      </div>
    </div>
  );
}

export default App;
  7. Run your React app
    8. npm start
  8. Experiement
    • Try changing the cursor’s size, color, or even shape. You can also render an emoji as your cursor for fun!
  9. Close the app 
    Press Ctrl + C on Mac/Linux/Windows or stop the process in GitHub Codespaces.
