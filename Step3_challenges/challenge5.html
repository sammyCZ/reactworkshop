  7. Add this code to index.js 
    // Basic Express server with persistent storage
const express = require("express");
const fs = require("fs");
const cors = require("cors");
const bodyParser = require("body-parser");
const app = express();
const PORT = 5000;

app.use(cors());
app.use(bodyParser.json());

// Path to your JSON storage
const DATA_FILE = "./data.json";

// Read the saved word from the file
const getSavedWord = () => {
  if (fs.existsSync(DATA_FILE)) {
    const data = fs.readFileSync(DATA_FILE);
    return JSON.parse(data).word || "";
  }
  return "";
};

// Write the word to the file
const saveWord = (word) => {
  fs.writeFileSync(DATA_FILE, JSON.stringify({ word }));
};

// GET the stored word
app.get("/word", (req, res) => {
  const word = getSavedWord();
  res.json({ word });
});

// POST a new word
app.post("/word", (req, res) => {
  const { word } = req.body;
  if (!word) {
    return res.status(400).json({ error: "No word provided" });
  }
  saveWord(word);
  res.json({ message: "Word saved!", word });
});

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});

  10. Install Axios in your React app: 
    npm install axios
  11. Update backend and frontend ports:
    • In server/index.js, change const PORT = 5000 to 5001.
    • In src/App.js, change the API URL port to 5001.
  12. Replace App.js with this code: 
    import React, { useState, useEffect } from "react";
import axios from "axios";
import "./App.css";

const API_URL = "http://localhost:5000/word";

function App() {
  const [inputWord, setInputWord] = useState("");
  const [savedWord, setSavedWord] = useState("");

  useEffect(() => {
    axios.get(API_URL)
      .then(res => setSavedWord(res.data.word))
      .catch(err => console.error("Error loading word:", err));
  }, []);

  const handleSubmit = async (e) => {
    e.preventDefault();
    try {
      const res = await axios.post(API_URL, { word: inputWord });
      setSavedWord(res.data.word);
      setInputWord("");
    } catch (err) {
      console.error("Error saving word:", err);
    }
  };

  return (
    <div className="App">
      <h1>üìù Persistent Word Saver</h1>

      <form onSubmit={handleSubmit}>
        <input
          type="text"
          value={inputWord}
          onChange={(e) => setInputWord(e.target.value)}
          placeholder="Type a word"
        />
        <button type="submit">Save Word</button>
      </form>

      <div className="saved">
        <h2>Saved Word:</h2>
        <p>{savedWord || "No word saved yet."}</p>
      </div>
    </div>
  );
}

export default App;
