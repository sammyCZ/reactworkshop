Challenge 6: Connect to an Arduino

  1. Goal: Connect React to an Arduino using the Web Serial API to display a message on a 7-segment display.
  2. Make sure you’re inside your React project: 
    cd my-app
  3. Create a components folder inside src: 
    mkdir src/components
  4. Create a new file inside components called ArduinoControl.js:
  5. Copy and paste the following code into ArduinoControl.js: 
    import React, { useState } from "react";

function ArduinoControl() {
  const [port, setPort] = useState(null);
  const [writer, setWriter] = useState(null);

  const connectToArduino = async () => {
    try {
      console.log("Requesting serial port...");
      const selectedPort = await navigator.serial.requestPort();
      console.log("Port selected:", selectedPort);
      await selectedPort.open({ baudRate: 9600 });
      console.log("Port opened!");

      const textEncoder = new TextEncoderStream();
      textEncoder.readable.pipeTo(selectedPort.writable);
      const portWriter = textEncoder.writable.getWriter();

      setPort(selectedPort);
      setWriter(portWriter);

      console.log("Connected to Arduino!");
    } catch (err) {
      console.error("Error connecting to Arduino:", err);
    }
  };

  const sendCommand = async (command) => {
    if (writer) {
      await writer.write(command + "\n");
      console.log(`Sent '${command}' to Arduino`);
    } else {
      console.log("No connection to Arduino");
    }
  };

  return (
    <div style={{ padding: "20px", fontFamily: "Arial" }}>
      <h1>The Arduino Controller</h1>
      {!port ? (
        <button onClick={connectToArduino}>Connect to Arduino</button>
      ) : (
        <div>
          <button onClick={() => sendCommand("start")} style={{ marginRight: "10px" }}>
            Start Message
          </button>
          <button onClick={() => sendCommand("stop")}>Stop Message</button>
        </div>
      )}
    </div>
  );
}

export default ArduinoControl;

  10. Copy and paste this Arduino sketch into a new file: 
    // Pin setup
int a = 7;
int b = 6;
int c = 5;
int d = 11;
int e = 10;
int f = 8;
int g = 9;
int dp = 4;

// Serial input
String inputString = "";
bool programRunning = false;

// ===== DISPLAY LETTERS =====
void displayY() { digitalWrite(f, HIGH); digitalWrite(g, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH); digitalWrite(d, HIGH); }
void displayA() { digitalWrite(f, HIGH); digitalWrite(e, HIGH); digitalWrite(g, HIGH); digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH); }
void displayL() { digitalWrite(e, HIGH); digitalWrite(f, HIGH); digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH); }
void displayO() { digitalWrite(f, HIGH); digitalWrite(e, HIGH); digitalWrite(d, HIGH); digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH); }
void clearDisplay() { digitalWrite(a, LOW); digitalWrite(b, LOW); digitalWrite(g, LOW); digitalWrite(c, LOW); digitalWrite(d, LOW); digitalWrite(e, LOW); digitalWrite(f, LOW); }

void showMessage() {
  clearDisplay();
  displayY();
  delay(2000);
  clearDisplay();
  displayA();
  delay(2000);
  clearDisplay();
  displayL();
  delay(2000);
  clearDisplay();
  displayO();
  delay(2000);
  clearDisplay();
}

void setup() {
  for (int i = 4; i <= 11; i++) {
    pinMode(i, OUTPUT);
  }
  Serial.begin(9600);
}

void loop() {
  if (Serial.available() > 0) {
    char received = (char)Serial.read();
    if (received == '\n') {
      inputString.trim();
      if (inputString == "start") {
        programRunning = true;
      }
      if (inputString == "stop") {
        programRunning = false;
        clearDisplay();
      }
      inputString = "";
    } else {
      inputString += received;
    }
  }

  if (programRunning) {
    showMessage();
    programRunning = false;
  }
}
